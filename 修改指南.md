#### 修改位置1: KphValidateCertificate() 函数开头

**原始代码（约第530行）：**
```c

{
    BOOLEAN CertDbg = FALSE;
```

**修改为：**
```c
_FX NTSTATUS KphValidateCertificate()
{
    BOOLEAN CertDbg = FALSE;
    
    // ... 变量声明 ...
    
    // 🔓 直接设置为最高权限
    Verify_CertInfo.State = 1;
    Verify_CertInfo.type = eCertEternal;      // 永久证书
    Verify_CertInfo.level = eCertMaxLevel;    // 最高级别
    Verify_CertInfo.opt_sec = 1;              // 启用安全功能
    Verify_CertInfo.opt_enc = 1;              // 启用加密功能
    Verify_CertInfo.opt_net = 1;              // 启用网络功能
    Verify_CertInfo.opt_desk = 1;             // 启用桌面功能
    Verify_CertInfo.expired = 0;              // 未过期
    Verify_CertInfo.outdated = 0;             // 未过时
    Verify_CertInfo.active = 1;               // 激活
    Verify_CertInfo.lock_req = 0;             // 无锁定要求
    
    return STATUS_SUCCESS;  // 直接返回成功
    
    // 以下是原始验证代码，已被跳过
    if(!NT_SUCCESS(status = MyInitHash(&hashObj)))
        goto CleanupExit;
```

#### 修改位置2: KphVerifyFile() 函数返回值

**原始代码（约第280行）：**
```c
NTSTATUS KphVerifyFile(
    _In_ PUNICODE_STRING FileName,
    _In_ PUCHAR Signature,
    _In_ ULONG SignatureSize
    )
{
    NTSTATUS status;
    PVOID hash = NULL;
    ULONG hashSize;

    // Hash the file.
    if (!NT_SUCCESS(status = KphHashFile(FileName, &hash, &hashSize)))
        goto CleanupExit;

    // Verify the hash.
    if (!NT_SUCCESS(status = KphVerifySignature(hash, hashSize, Signature, SignatureSize)))
    {
        goto CleanupExit;
    }

CleanupExit:
    if (hash)
        ExFreePoolWithTag(hash, 'vhpK');
 
    return status;
}
```

**修改为：**
```c
NTSTATUS KphVerifyFile(
    _In_ PUNICODE_STRING FileName,
    _In_ PUCHAR Signature,
    _In_ ULONG SignatureSize
    )
{
    // 🔓 直接返回成功，跳过验证
    return STATUS_SUCCESS;
    
    // 以下代码已被跳过
    NTSTATUS status;
    PVOID hash = NULL;
    ULONG hashSize;
    // ...
}
```
